274  /**
275   * Calculate elapsed time.
276   *
277   * @return the long
278   */
279  private long calculateElapsedTime() {
280    elapsedTime = 0;
281
282    if (getEndTime() != null && getStartTime() != null) {
283      Instant end = getEndTime().toInstant();
284      Instant start = getStartTime().toInstant();
285      elapsedTime = ChronoUnit.MINUTES.between(start, end);
286      elapsedHours = (double) (elapsedTime / 60d);
287    }
288
289    return elapsedTime;
290  }
291

299  }

305   */
306  public double getElapsedHours() {
307    if (elapsedHours <= 0) {
308      calculateElapsedTime();
309    }
310
311    return elapsedHours;
312  }

346
347  /**
348   * @see java.lang.Comparable#compareTo(java.lang.Object)
349   */
350  public int compareTo(TaskViewModel obj) {
351    int finalOrder = 0;
352
353    finalOrder = compareStrings(wbsName, obj.getWbsName());
354
355    if (finalOrder == 0) {
356      finalOrder = compareStrings(activityName, obj.getActivityName());
357    }
358
359    if (finalOrder == 0) {
360      finalOrder = Double.valueOf(elapsedHours).compareTo(Double.valueOf(obj.getElapsedHours()));
361    }
362
363    return finalOrder;
364  }
365
366  // TODO - move to utility class
367  /**
368   * Compare two Strings.
369   *
370   * @param left the left
371   * @param right the right
372   * @return the int
373   */
374  private int compareStrings(String left, String right) {
375    int finalOrder = 0;
376
377    if (left == null && right != null) {
378      finalOrder = -1;
379    } else if (left != null && right == null) {
380      finalOrder = 1;
381    } else {
382      finalOrder = left.compareToIgnoreCase(right);
383    }
384
385    return finalOrder;
386  }
